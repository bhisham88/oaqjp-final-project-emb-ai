"""
This is server handler file for emotion_detector functionality
that handles user requests coming to /emotionDetector endpoint
"""

from flask import Flask

import emotion_detection

app = Flask(__name__)

# USER_MESSAGE = "I love my life"
USER_MESSAGE = "I think I am having fun"
# USER_MESSAGE = "I hate working long hours"
# USER_MESSAGE = "" # for error handling purpose

@app.route("/emotionDetector")
def emotion_detector():
    """
    This route/function handles user requests coming to /emotionDetector endpoint
    """
    response = emotion_detection.emotion_detector(USER_MESSAGE)

    # Null Check
    is_response_from_emotion_detector_none = all(values is None for values in response.values())
    if is_response_from_emotion_detector_none:
        return ("--------------------------------\n"
                "Invalid text! Please try again!\n"
                "--------------------------------\n")

    anger = str(response.get("anger"))
    disgust = str(response.get("disgust"))
    fear = str(response.get("fear"))
    joy = str(response.get("joy"))
    sadness = str(response.get("sadness"))
    dominant_emotion = str(response.get("dominant_emotion"))

    response_message = (f"For the given statement, the system response is 'anger': {anger}"
                        f", 'disgust': {disgust}, 'fear': {fear}"
                        f", 'joy': {joy} and 'sadness': {sadness}"
                        f". The dominant emotion is {dominant_emotion}.")
    return response_message


if __name__ == "__main__":
    app.run(debug=True)
